#2 Create a random string of length 3 from a given set of chars. 

    let setOfChars = `asdfjkl;qwertyuiop12348098-09!@#$@#!%^&`;
    [...new Array(3)].map(function(){
        let bias = 9;
        let randomIndex = Math.floor(Math.random()*bias);
        let getChar = this[randomIndex]
        return getChar;
    },setOfChars).join('');

#4 Pass an invoked function with argument to a function's parameter

    function _driver(fn = f()){
        let a = 4;
        let b = fn;
        return a+b;
    }
    function _helper (num){
        return num;
    }
    _driver(_helper(44)); //48

#5 Swap using destructuring

    let arr = [1,2,3,4,5];
    [arr[0], arr[1]] = [arr[1],arr[0]];
    console.log(arr); // (5) [2, 1, 3, 4, 5]
