    let name = "Raj";
    function modifyName(new_name){
    	 name = new_name;
    	 return name; 
    }
    function concatName(_name){
    	 name =  _name + " Babbar";
    	 return name; 
    }
    console.log(name);
    modifyName("Raaz");
    console.log(name);
    concatName("Raj")
    console.log(name);

I don't like *name* variable to be used by any other script or function and I want to be this way, 
1. name variable shall only be used by modifyName. i.e. it shall be private to modifyName only.
2. name variable must be accessible outside of its scope.

**Apply closure** 

Here updateName, alias modifyName. 

    const updateName = (function(){
        var name = "Iyer"; //has scope inside this anonymous function
        return function(new_name){ // <== this is the closure function. It persists the state of its lexical scope variable (here its "name") on each call even its closed 
            name = new_name + " " + name;
            return name;
        }
    })();
    
    name = "R_A_J"; // global scope and it has nothing to do with the name var inside updateName
    console.log(updateName("Venugopal"));
    console.log(updateName("Muthuswami"));
    console.log(updateName("Chinnaswami"));
    console.log(updateName("Parambaturr"));
    console.log(updateName("Parampeel"));
    console.log(updateName("Yekya"));

Output

> Venugopal Iyer 

> Muthuswami Venugopal Iyer 

> Chinnaswami Muthuswami Venugopal Iyer 

> Parambaturr Chinnaswami Muthuswami Venugopal Iyer

> Parampeel Parambaturr Chinnaswami Muthuswami Venugopal Iyer 

> Yekya Parampeel Parambaturr Chinnaswami Muthuswami Venugopal Iyer


1. name variable is now only used by updateName. i.e. its private to updateName only.
2. name variable is now accessible outside of its scope by indirectly calling the updateName function.

Further, the initial "Iyer" value is preserved even after consecutive calls to the closure.
