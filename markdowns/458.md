```js
function power(a, b, mod) {
  let res = 1;
  a %= mod;
  while (b > 0) {
    if (b % 2 === 1) res = (res * a) % mod;
    a = (a * a) % mod;
    b = Math.floor(b / 2);
  }
  return res;
}
```

### âœ… Prime Sieve:
```js
function sieve(n) {
  const isPrime = Array(n + 1).fill(true);
  isPrime[0] = isPrime[1] = false;
  for (let i = 2; i * i <= n; i++) {
    if (isPrime[i]) {
      for (let j = i * i; j <= n; j += i) isPrime[j] = false;
    }
  }
  return isPrime;
}
```

```js
  constructor() {
    this.heap = [];
  }
  insert(val) {
    this.heap.push(val);
    this.bubbleUp();
  }
  extractMin() {
    if (this.heap.length === 1) return this.heap.pop();
    const min = this.heap[0];
    this.heap[0] = this.heap.pop();
    this.bubbleDown();
    return min;
  }
```
